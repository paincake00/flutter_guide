
```go
// Арифметические операторы
a := 10
b := 3

sum := a + b       // Сложение: 13
diff := a - b      // Вычитание: 7
product := a * b   // Умножение: 30
quotient := a / b  // Деление: 3 (целочисленное)
remainder := a % b // Остаток: 1

// Операторы сравнения
equal := a == b     // false
notEqual := a != b  // true
greater := a > b    // true
less := a < b       // false
greaterEqual := a >= b // true
lessEqual := a <= b   // false

// Логические операторы
andResult := true && false  // false
orResult := true || false   // true
notResult := !true          // false

// Побитовые операторы
bitwiseAND := a & b      // 2 (10 & 3 = 2)
bitwiseOR := a | b       // 11 (10 | 3 = 11)
bitwiseXOR := a ^ b      // 9 (10 ^ 3 = 9)
leftShift := a << 1      // 20 (10 << 1 = 20)
rightShift := a >> 1     // 5 (10 >> 1 = 5)

// Операторы присваивания
x := 10
x += 5   // x = x + 5
x -= 3   // x = x - 3
x *= 2   // x = x * 2
x /= 4   // x = x / 4
x %= 3   // x = x % 3
```
